# .github/workflows/vendor-integration.yml
# Live vendor-API integration for the official plugins. These tests perform real
# network calls and require credentials provided as repository secrets; each
# vendor step is skipped when its secret is absent, so the workflow is safe to
# run without full credentials.
name: vendor-integration

on:
  workflow_dispatch:
  schedule:
    - cron: '0 4 * * 1'   # weekly

permissions:
  contents: read

# ── Concurrency ────────────────────────────────────────────────
concurrency:
  group: vendor-integration-${{ github.ref }}
  cancel-in-progress: true

jobs:
  vendor:
    runs-on: ubuntu-latest
    timeout-minutes: 15
    # A real Redis backs the storage plugin's Layer-B integration; when present
    # the redis plugin earns VERIFIED, otherwise it is recorded BLOCKED honestly.
    services:
      redis:
        image: redis:7-alpine
        ports:
          - 6379:6379
        options: >-
          --health-cmd "redis-cli ping"
          --health-interval 5s
          --health-timeout 3s
          --health-retries 5
    # Secrets cannot be referenced in step-level `if:` conditions, so they are
    # mapped to env at the job level and the gates below test `env.*` instead.
    env:
      REDIS_HOST: 127.0.0.1
      REDIS_PORT: '6379'
      SENDGRID_API_KEY: ${{ secrets.SENDGRID_API_KEY }}
      STRIPE_API_KEY: ${{ secrets.STRIPE_API_KEY }}
      TWILIO_ACCOUNT_SID: ${{ secrets.TWILIO_ACCOUNT_SID }}
      TWILIO_AUTH_TOKEN: ${{ secrets.TWILIO_AUTH_TOKEN }}
      S3_BUCKET: ${{ secrets.S3_BUCKET }}
      S3_REGION: ${{ secrets.S3_REGION }}
      S3_ACCESS_KEY_ID: ${{ secrets.S3_ACCESS_KEY_ID }}
      S3_SECRET_ACCESS_KEY: ${{ secrets.S3_SECRET_ACCESS_KEY }}
      R2_ACCOUNT_ID: ${{ secrets.R2_ACCOUNT_ID }}
      R2_BUCKET: ${{ secrets.R2_BUCKET }}
      R2_ACCESS_KEY_ID: ${{ secrets.R2_ACCESS_KEY_ID }}
      R2_SECRET_ACCESS_KEY: ${{ secrets.R2_SECRET_ACCESS_KEY }}
      AUTH0_DOMAIN: ${{ secrets.AUTH0_DOMAIN }}
      AUTH0_CLIENT_ID: ${{ secrets.AUTH0_CLIENT_ID }}
      AUTH0_CLIENT_SECRET: ${{ secrets.AUTH0_CLIENT_SECRET }}
      AUTH0_AUDIENCE: ${{ secrets.AUTH0_AUDIENCE }}
      SUPABASE_URL: ${{ secrets.SUPABASE_URL }}
      SUPABASE_KEY: ${{ secrets.SUPABASE_KEY }}
      SUPABASE_BUCKET: ${{ secrets.SUPABASE_BUCKET }}
      GCS_BUCKET: ${{ secrets.GCS_BUCKET }}
      GCS_PROJECT_ID: ${{ secrets.GCS_PROJECT_ID }}
      AZURE_STORAGE_CONNECTION_STRING: ${{ secrets.AZURE_STORAGE_CONNECTION_STRING }}
      AZURE_STORAGE_CONTAINER: ${{ secrets.AZURE_STORAGE_CONTAINER }}
      STREETJS_B2_ENDPOINT: ${{ secrets.STREETJS_B2_ENDPOINT }}
      STREETJS_B2_BUCKET: ${{ secrets.STREETJS_B2_BUCKET }}
      STREETJS_B2_KEY_ID: ${{ secrets.STREETJS_B2_KEY_ID }}
      STREETJS_B2_APPLICATION_KEY: ${{ secrets.STREETJS_B2_APPLICATION_KEY }}
      GCS_SERVICE_ACCOUNT_JSON: ${{ secrets.GCS_SERVICE_ACCOUNT_JSON }}
    steps:
      - uses: actions/checkout@df4cb1c069e1874edd31b4311f1884172cec0e10 # v6.0.3
        with:
          persist-credentials: false
      - uses: ./.github/actions/setup
      - name: Build core
        run: npm run build -w packages/core

      - name: Build redis plugin (for its Layer-B integration)
        run: npm run build -w packages/plugin-redis

      # Each step builds the vendor request via the plugin client and performs a
      # minimal real call, asserting an authenticated (non-401/403) response.
      - name: SendGrid (scopes check)
        if: ${{ env.SENDGRID_API_KEY != '' }}
        run: node scripts/vendor/check-sendgrid.mjs

      - name: Stripe (balance check)
        if: ${{ env.STRIPE_API_KEY != '' }}
        run: node scripts/vendor/check-stripe.mjs

      # ── Cloud storage providers (Supabase, GCS, Azure, Backblaze B2) ────────
      # packages/storage's own test suite (dist/tests/*.test.js) already
      # implements an honest credential-based skip policy per provider (see
      # non-s3-integration.test.ts / s3-integration.test.ts): a provider's live
      # round-trip only runs when ALL of its env vars are present, otherwise it
      # is reported as skipped (never as a false pass). No mocking, ever.
      # Ordered before Auth0 so an unrelated Auth0 credential/tenant issue never
      # blocks these independent checks (each step here is independently gated
      # and does not depend on Auth0 succeeding).
      - name: Build storage package (for its live provider integration tests)
        run: npm run build -w packages/storage

      # The Supabase/GCS/Azure/Backblaze drivers resolve their vendor SDK via a
      # lazy dynamic import() (never at module load), so the package has zero
      # hard SDK dependencies. Without the SDK installed, a driver with valid
      # credentials still can't run its live round-trip and is honestly
      # reported as "passed-with-skip" (Requirement 27.4) rather than a false
      # failure — but that means credentials alone don't prove the driver
      # actually works end-to-end. Installing the real, optional peer SDKs here
      # lets the tests exercise the genuine live round-trip whenever credentials
      # are present.
      # `--prefix packages/storage --package-lock=false` (not `--no-save`, not
      # the root `-w` form, not a plain `working-directory:` + bare `npm
      # install`) is required here. Every other shape was tried and each
      # silently no-ops or fails:
      #   - the root `-w packages/storage` form hits a real npm 10.8.2 Arborist
      #     bug ("Cannot read properties of undefined (reading 'spec')") when
      #     saving the ideal tree for a workspace member;
      #   - `--no-save` silently no-ops against an already-reconciled workspace
      #     lockfile (reports success but installs nothing);
      #   - running plain `npm install <pkgs> --package-lock=false` with cwd set
      #     to `packages/storage/` (whether via `working-directory:` or a shell
      #     `cd`) still resolves the ambient workspace root from `package.json`'s
      #     `workspaces` field and silently no-ops with `npm warn workspaces
      #     @streetjs/storage in filter set, but no workspace folder present` —
      #     it never touches `packages/storage/node_modules/` at all.
      # `--prefix` is the one flag that actually overrides where npm resolves
      # its install root, bypassing the workspace auto-detection entirely while
      # still skipping the lockfile save. This never touches the committed repo
      # since the job never pushes.
      - name: Install optional cloud-provider SDKs (for real live round-trips)
        run: npm install @supabase/supabase-js @google-cloud/storage @azure/storage-blob --package-lock=false --prefix packages/storage

      - name: Materialize GCS service-account key file
        if: ${{ env.GCS_SERVICE_ACCOUNT_JSON != '' }}
        run: |
          echo "$GCS_SERVICE_ACCOUNT_JSON" > "$RUNNER_TEMP/gcs-service-account.json"
          echo "GOOGLE_APPLICATION_CREDENTIALS=$RUNNER_TEMP/gcs-service-account.json" >> "$GITHUB_ENV"

      - name: Cloud storage providers (Supabase / GCS / Azure / Backblaze B2)
        run: npm run test -w packages/storage

      - name: Auth0 (client-credentials token)
        if: ${{ env.AUTH0_DOMAIN != '' }}
        run: node scripts/vendor/check-auth0.mjs

      # Layer B: run every official plugin's REAL backing-service integration
      # through CommandRunner, emitting one plugin.<id>.artifact.json each
      # (pass result + plugin id + ISO-8601 timestamp). A plugin whose service
      # or test account is absent is recorded BLOCKED with its missing
      # prerequisite id — never mocked (Requirement 5.9).
      - name: Plugin integration verification (Layer B)
        run: node scripts/plugins/verify.mjs

      - name: Upload plugin verification artifacts
        if: ${{ always() }}
        uses: actions/upload-artifact@043fb46d1a93c77aae656e7c1c64a875d1fc6a0a # v7.0.1
        with:
          name: plugin-verification-artifacts
          path: verification-artifacts/plugins/
          if-no-files-found: warn
          retention-days: 30
